#3.Pyramid star pattern
n = int(input('Enter a number:'))
for i in range(1,n+1,2):
    for j in range((n-i)//2):
        print(' ',end='')
    for k in range(i):
        print('*',end='')
    print()

#9. Number pyramid pattern
n = int(input('Enter a number:'))
for i in range(1,n+1):
    print(' ' * (n - i), end='')
    for j in range(1, i + 1):
        print(j, end=" ")
    print()
